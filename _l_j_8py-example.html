1 # A simple script for running a Lennard-Jones simulation.
2 
3 # Statistics such as energy, (instantaneous) temperature, and (instantaneous) pressure are
4 # output to "LJ-example.npz", and coordinates are output to "LJ-example.xyz".
5 # These xyz files can be read by VMD; a file named "LJ-example.tcl" is also created for use with vmd.
6 # You can then run 'vmd -e LJ-example.tcl LJ-example.xyz' to see the simulation.
7 
8 # This was written for Python 3.4+; it may take some small effort to use a lower version of Python.
9 
10 #---------------------------------------------------------------------------------------------------
11 from math import pi, sqrt
12 import numpy as np
13 import pyparm.d3 as sim # or d2 for 2D simulation
14 import pyparm.util as util
15 
16 #---------------------------------------------------------------------------------------------------
17 # Parameters
18 
19 # filenames
20 data_file = 'LJ-example.npz'
21 tcl_file = 'LJ-example.tcl'
22 xyz_file = 'LJ-example.xyz'
23 
24 # system parameters
25 N = 40
26 N1, N2 = (N//2), (N - (N//2))
27 T0 = 1
28 phi = 0.3 # "packing fraction", i.e. (volume of spheres) / (volume of box)
29 sigmas = ([1] * N1) + ([1.4] * N2)
30 
31 # simulation parameters
32 dt = 0.001
33 total_time = 100
34 data_n = 1000
35 xyz_n = 1000
36 print_n = 20
37 
38 total_steps = int(total_time / dt + 0.5)
39 xyz_dt = float(total_time) / xyz_n
40 data_dt = float(total_time) / data_n
41 print_dt = float(total_time) / print_n
42 
43 # purely calculated
44 d = sim.NDIM # number of dimensions
45 volumes = [s**d * pi/(2*d) for s in sigmas]
46 masses = [v / min(volumes) for v in volumes] # mass proportional to volume, but 1 for the smallest
47 L = (sum(volumes) / phi) ** (1.0/d) # length of each side of the box
48 
49 #---------------------------------------------------------------------------------------------------
50 # Setting up the simulation
51 box = sim.OriginBox(L)
52 atoms = sim.AtomVec(masses)
53 neighbors = sim.NeighborList(box, atoms, 0.4) # the neighborlist, for keeping track of what atoms are near what other atoms
54 LJ = sim.LJgroup(atoms, neighbors)
55 collec = sim.CollectionVerlet(box, atoms, dt, [LJ], [neighbors]) # the integrator
56 # We use a simple velocity-verlet integrator, which is time-reversible and NVE ensemble
57 # i.e., it preserves number of atoms, volume of box, and energy
58 
59 #---------------------------------------------------------------------------------------------------
60 # Initial Conditions
61 # Now we have created our Atom, but we need to add our atoms to it. We do that in a way that prevents overlap
62 
63 E0 = 0
64 for a,s in zip(atoms, sigmas):
65  E = E0 + 10
66  a.v = sim.randVec() # from a gaussian distribution
67  LJ.add(sim.LJatom(1, s, a))
68  while E > E0 + 0.1:
69  a.x = box.randLoc()
70  neighbors.update_list()
71  E = LJ.energy(box)
72  E0 = E
73 
74 collec.resetcomv() # subtract center-of-mass velocity from all particles
75 collec.scaleVelocitiesT(T0) # scale all velocities to get an instantaneous temperature T = T0, at least at the beginning
76

88 data_arrays['t'] = np.zeros((data_n,))

91  """Take each measurement in data_functions at time 'time', and store it in data_arrays at index idx"""
92  data_arrays['t'][idx] = time
93  for k, f in data_functions.items():
94  data_arrays[k][idx] = f()
95 
96 def write_data(idx):
97  delim = '\t'
98  keys = sorted(data_arrays.keys())
99  header = delim.join(keys)
100  data = {k:data_arrays[k][:idx+1] for k in keys}
101  np.savez_compressed(data_file, **data)

108  # print the current line to file

114  print(e, *x, file=f)

135 
136 with open(tcl_file, 'w') as f:
137  print(tcl_str, file=f)
